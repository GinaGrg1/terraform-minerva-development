# In rg.tfvars file: 
resource_group_data = {
    development_centralus_rg_01 = {
        enabled = true
        name = "Gurung-Test"
        location = "CentralUS"
    }
    development_centralus_rg_02 = {
        enabled = true
        name = "Gurung-Test-2"
        location = "CentralUs"
    }
}

# In variables.tf file you will have
variable "resource_group_data" {
    type = map(object({
        #Required
        enabled = bool
        location = string
        name = string

        # Optional
        managed_by = optional(string)
        tags = optional(string)
    }))
    default = {}
}
